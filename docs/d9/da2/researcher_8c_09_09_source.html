111 QGeoPath Researcher::plotGeopathProfile(const QGeoPath& path)
112 {
113 QGeoPath ret;
114 for(QGeoCoordinate point : path.path())
115 {
116 if(ret.size())
117 {
118 QGeoCoordinate previous = ret.coordinateAt(ret.size() - 1);
119 auto azimuth = static_cast<float>(previous.azimuthTo(point));
120 auto distance_from_previous = static_cast<float>(previous.distanceTo(point));
121 float distance = SCAN_STEP;
122 QGeoCoordinate previous_delta = previous;
123
124 while(distance < distance_from_previous)
125 {
126 QGeoCoordinate delta_point = previous.atDistanceAndAzimuth(distance, azimuth);
127 delta_point.setAltitude(DEM::elevation(delta_point.latitude(), delta_point.longitude()));
128
129 if(previous_delta.altitude() != delta_point.altitude())
130 {
131 if(previous_delta.altitude() > delta_point.altitude())
132 ret.addCoordinate(previous_delta);
133 else
134 ret.addCoordinate(delta_point);
135 }
136
137 previous_delta = delta_point;
138 distance += SCAN_STEP;
139 }
140 }
141
142 point.setAltitude(DEM::elevation(point.latitude(), point.longitude()));
143 ret.addCoordinate(point);
144 }
