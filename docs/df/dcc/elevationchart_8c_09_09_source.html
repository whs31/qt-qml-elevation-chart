137 void ElevationChartItem::updateBounds() noexcept
138 {
139 if(not route().valid())
140 return;
141
142 setBounds({ m_profile.back().distance(), 0 });
143 auto path = route().toGeoPath().path();
144 for(const auto& coordinate : path)
145 if(coordinate.altitude() > bounds().y())
146 setBounds({bounds().x(), static_cast<float>(coordinate.altitude())});
147
148 if(shrinkMode() == ShrinkToProfileHeight)
149 setBounds({bounds().x(), std::max(bounds().y(), std::max_element(m_profile.cbegin(), m_profile.cend(),
150 [](const ElevationPoint& a, const ElevationPoint& b){ return a.elevation() < b.elevation();
151 })->elevation())});
152
153 this->updateMetrics();
154 this->update();
155 }

157 void ElevationChartItem::updateMetrics() noexcept
158 {
159 bool matching_metrics = true;
160
161 QGeoPath correct_path;
162 QGeoPath route_path = route().toGeoPath();
163 if(route_path.isEmpty())
164 return;
165
166 //bool allow_individual_speeds = not qFuzzyCompare(route().at(0).velocity(), 0);
167 auto velocities = route().velocities();
168
169 correct_path.addCoordinate(route_path.path().first());
170 for(int i = 1; i < route_path.path().size(); i++)
171 {
172 const bool fallback = velocities.at(i) == 0;
173 const auto dx = static_cast<float>(route_path.path().at(i).distanceTo(route_path.path().at(i - 1)));
174 const auto dy = static_cast<float>(route_path.path().at(i).altitude() - correct_path.path().at(i - 1).altitude());
175 const float dy_min = fallback ? metrics().rateOfDescend() * dx / metrics().fallbackVelocity() : metrics().rateOfDescend() * dx / velocities.at(i);
176 const float dy_max = fallback ? metrics().rateOfClimb() * dx / metrics().fallbackVelocity() : metrics().rateOfClimb() * dx / velocities.at(i);
177
178 auto result = route_path.path().at(i);
179
180 // correct case
181 if((dy >= 0 and dy < dy_max) or (dy <= 0 and abs(dy) < dy_min))
182 {
183 correct_path.addCoordinate(result);
184 continue;
185 }
186
187 // climbs too fast
188 else if(dy > 0 and dy > dy_max)
189 {
190 result.setAltitude(correct_path.path().at(i - 1).altitude() + dy_max);
191 matching_metrics = false;
192 }
193
194 // descends too fast
195 else if(dy < 0 and abs(dy) > dy_min)
196 {
197 result.setAltitude(correct_path.path().at(i - 1).altitude() - dy_min);
198 matching_metrics = false;
199 }
200
201 correct_path.addCoordinate(result);
202 }
203
204 if(not matching_metrics)
205 {
206 setMatchingMetrics(false);
207 m_metrics_path = correct_path;
208 }
209 else
210 setMatchingMetrics(true);
211 }

224 void ElevationChartItem::handleProfileNode() noexcept
225 {
226 vector<QSGGeometry::Point2D> gl;
227 for(const auto& point : m_profile)
228 {
229 gl.push_back({toPixelX(point.distance(), bounds().x()), static_cast<float>(height())});
230 gl.push_back(toPixel(point.distance(), point.elevation(), bounds().x(), bounds().y()));
231 }
232
233 m_profile_node->geometry()->allocate(static_cast<int>(gl.size()));
234
235 for(size_t i = 0; i < gl.size(); i++)
236 m_profile_node->geometry()->vertexDataAsPoint2D()[i] = gl.at(i);
237 }
