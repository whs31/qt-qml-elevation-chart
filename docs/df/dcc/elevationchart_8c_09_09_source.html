51 QSGNode* ElevationChartItem::updatePaintNode(QSGNode* old_node, QQuickItem::UpdatePaintNodeData* unused)
52 {
53 (void)unused;
54
55 if(old_node == nullptr)
56 {
57 old_node = new QSGNode;
58
59 m_background_node = new QSGGeometryNode;
60 auto* background_material = new QSGFlatColorMaterial;
61 background_material->setColor(palette().background());
62 m_background_node->setMaterial(background_material);
63 auto* background_geometry = new QSGGeometry(QSGGeometry::defaultAttributes_Point2D(), 0, 0, QSGGeometry::UnsignedIntType);
64 background_geometry->setDrawingMode(QSGGeometry::DrawTriangles);
65 m_background_node->setGeometry(background_geometry);
66 m_background_node->setFlags(QSGNode::OwnsGeometry | QSGNode::OwnsMaterial);
67
68 m_profile_node = new QSGGeometryNode;
69 auto* profile_material = new QSGFlatColorMaterial;
70 profile_material->setColor(palette().overlay());
71 m_profile_node->setMaterial(profile_material);
72 auto* profile_geometry = new QSGGeometry(QSGGeometry::defaultAttributes_Point2D(), 0, 0, QSGGeometry::UnsignedIntType);
73 profile_geometry->setDrawingMode(GL_QUAD_STRIP);
74 m_profile_node->setGeometry(profile_geometry);
75 m_profile_node->setFlags(QSGNode::OwnsGeometry | QSGNode::OwnsMaterial);
76
77 m_route_node = new QSGGeometryNode;
78 auto* route_material = new QSGFlatColorMaterial;
79 route_material->setColor(palette().accent());
80 m_route_node->setMaterial(route_material);
81 auto* route_geometry = new QSGGeometry(QSGGeometry::defaultAttributes_Point2D(), 0, 0, QSGGeometry::UnsignedIntType);
82 route_geometry->setDrawingMode(QSGGeometry::DrawLineStrip);
83 route_geometry->setLineWidth(ROUTE_LINE_WIDTH);
84 m_route_node->setGeometry(route_geometry);
85 m_route_node->setFlags(QSGNode::OwnsGeometry | QSGNode::OwnsMaterial);
86
87 old_node->appendChildNode(m_background_node);
88 old_node->appendChildNode(m_profile_node);
89 old_node->appendChildNode(m_route_node);
90 }
91
92 if(m_require_recolor)
93 {
94 dynamic_cast<QSGFlatColorMaterial*>(m_background_node->material())->setColor(palette().background());
95 dynamic_cast<QSGFlatColorMaterial*>(m_profile_node->material())->setColor(palette().overlay());
96 dynamic_cast<QSGFlatColorMaterial*>(m_route_node->material())->setColor(palette().accent());
97 }
98
99 this->handleBackgroundNode();
100 this->handleProfileNode();
101 this->handleRouteNode();
102
103 if(m_require_recolor)
104 {
105 m_background_node->markDirty(QSGNode::DirtyGeometry | QSGNode::DirtyMaterial);
106 m_profile_node->markDirty(QSGNode::DirtyGeometry | QSGNode::DirtyMaterial);
107 m_route_node->markDirty(QSGNode::DirtyGeometry | QSGNode::DirtyMaterial);
108 this->fulfillRecolor();
109 }
110 else
111 {
112 m_background_node->markDirty(QSGNode::DirtyGeometry);
113 m_profile_node->markDirty(QSGNode::DirtyGeometry);
114 m_route_node->markDirty(QSGNode::DirtyGeometry);
115 }
116
117 old_node->markDirty(QSGNode::DirtyGeometry | QSGNode::DirtyMaterial);
118 return old_node;
119 }

141 void ElevationChartItem::updateBounds() noexcept
142 {
143 if(not route().valid())
144 return;
145
146 setBounds({ m_profile.back().distance(), 0 });
147 auto path = route().toGeoPath().path();
148 for(const auto& coordinate : path)
149 if(coordinate.altitude() > bounds().y())
150 setBounds({bounds().x(), static_cast<float>(coordinate.altitude())});
151
152 if(shrinkMode() == ShrinkToProfileHeight)
153 setBounds({bounds().x(), std::max(bounds().y(), std::max_element(m_profile.cbegin(), m_profile.cend(),
154 [](const ElevationPoint& a, const ElevationPoint& b){ return a.elevation() < b.elevation();
155 })->elevation())});
156
157 this->update();
158 }

160 void ElevationChartItem::handleBackgroundNode() noexcept
161 {
162 m_background_node->geometry()->allocate(6);
163 m_background_node->geometry()->vertexDataAsPoint2D()[0].set(0, 0);
164 m_background_node->geometry()->vertexDataAsPoint2D()[1].set(static_cast<float>(width()), 0);
165 m_background_node->geometry()->vertexDataAsPoint2D()[2].set(static_cast<float>(width()), static_cast<float>(height()));
166 m_background_node->geometry()->vertexDataAsPoint2D()[3].set(static_cast<float>(width()), static_cast<float>(height()));
167 m_background_node->geometry()->vertexDataAsPoint2D()[4].set(0, static_cast<float>(height()));
168 m_background_node->geometry()->vertexDataAsPoint2D()[5].set(0, 0);
169 }

171 void ElevationChartItem::handleProfileNode() noexcept
172 {
173 vector<QSGGeometry::Point2D> gl;
174 for(const auto& point : m_profile)
175 {
176 gl.push_back({toPixelX(point.distance(), bounds().x()), static_cast<float>(height())});
177 gl.push_back(toPixel(point.distance(), point.elevation(), bounds().x(), bounds().y()));
178 }
179
180 m_profile_node->geometry()->allocate(static_cast<int>(gl.size()));
181
182 for(size_t i = 0; i < gl.size(); i++)
183 m_profile_node->geometry()->vertexDataAsPoint2D()[i] = gl.at(i);
184 }

186 void ElevationChartItem::handleRouteNode() noexcept
187 {
188 vector<ElevationPoint> t_route = route().toElevationGraph();
189 vector<QSGGeometry::Point2D> gl;
190
191 for(const auto& point : t_route)
192 gl.push_back(toPixel(point.distance(), point.elevation(), bounds().x(), bounds().y()));
193
194 m_route_node->geometry()->allocate(static_cast<int>(gl.size()));
195 for(size_t i = 0; i < gl.size(); i++)
196 m_route_node->geometry()->vertexDataAsPoint2D()[i] = gl.at(i);
197 }
