151 void ElevationChartItem::updateBounds() noexcept
152 {
153 if(not route().valid())
154 return;
155
156 setBounds({ m_profile.back().distance(), 0 });
157 auto path = route().toGeoPath().path();
158 for(const auto& coordinate : path)
159 if(coordinate.altitude() > bounds().y())
160 setBounds({bounds().x(), static_cast<float>(coordinate.altitude())});
161
162 if(shrinkMode() == ShrinkToProfileHeight)
163 setBounds({bounds().x(), std::max(bounds().y(), std::max_element(m_profile.cbegin(), m_profile.cend(),
164 [](const ElevationPoint& a, const ElevationPoint& b){ return a.elevation() < b.elevation();
165 })->elevation())});
166
167 this->updateMetrics();
168 this->update();
169 }

171 void ElevationChartItem::updateMetrics() noexcept
172 {
173 bool matching_metrics = true;
174
175 QGeoPath correct_path;
176 QGeoPath route_path = route().toGeoPath();
177 if(route_path.isEmpty())
178 return;
179
180 //bool allow_individual_speeds = not qFuzzyCompare(route().at(0).velocity(), 0);
181 auto velocities = route().velocities();
182
183 correct_path.addCoordinate(route_path.path().first());
184 for(int i = 1; i < route_path.path().size(); i++)
185 {
186 const bool fallback = velocities.at(i) == 0;
187 const auto dx = static_cast<float>(route_path.path().at(i).distanceTo(route_path.path().at(i - 1)));
188 const auto dy = static_cast<float>(route_path.path().at(i).altitude() - correct_path.path().at(i - 1).altitude());
189 const float dy_min = fallback ? metrics().rateOfDescend() * dx / metrics().fallbackVelocity() : metrics().rateOfDescend() * dx / velocities.at(i);
190 const float dy_max = fallback ? metrics().rateOfClimb() * dx / metrics().fallbackVelocity() : metrics().rateOfClimb() * dx / velocities.at(i);
191
192 auto result = route_path.path().at(i);
193
194 // correct case
195 if((dy >= 0 and dy < dy_max) or (dy <= 0 and abs(dy) < dy_min))
196 {
197 correct_path.addCoordinate(result);
198 continue;
199 }
200
201 // climbs too fast
202 else if(dy > 0 and dy > dy_max)
203 {
204 result.setAltitude(correct_path.path().at(i - 1).altitude() + dy_max);
205 matching_metrics = false;
206 }
207
208 // descends too fast
209 else if(dy < 0 and abs(dy) > dy_min)
210 {
211 result.setAltitude(correct_path.path().at(i - 1).altitude() - dy_min);
212 matching_metrics = false;
213 }
214
215 correct_path.addCoordinate(result);
216 }
217
218 if(not matching_metrics)
219 {
220 setMatchingMetrics(false);
221 m_metrics_path = correct_path;
222 }
223 else
224 setMatchingMetrics(true);
225 }

266 void ElevationChartItem::handleMetricsNode() noexcept
267 {
268 if(matchingMetrics())
269 {
270 /*
271 * окей, это не совсем очевидная херня, поэтому я прокомментирую будущим поколениям.
272 * если мы не будем очищать буфер ноды, то при отсутствии вызова обновления метрик в
273 * соответствии с маршрутом мы получим невалидные данные. если эта нода была бы отдельным
274 * компонентом в qml, все решилось бы visible: !matchingMetrics, но внутри одного итема
275 * единственный варик это вручную чистить буфер вот таким вызовом.
276 */
277 m_metrics_point_node->geometry()->allocate(0);
278 m_metrics_node->geometry()->allocate(0);
279 return;
280 }
281
282 vector<ElevationPoint> t_route;
283 auto prev_coord = m_metrics_path.path().front();
284 float distance = 0;
285 for(const auto& point : m_metrics_path.path())
286 {
287 distance += static_cast<float>(point.distanceTo(prev_coord));
288 t_route.emplace_back(distance, point.altitude());
289 prev_coord = point;
290 }
291
292 vector<QSGGeometry::Point2D> gl;
293
294 for(const auto& point : t_route)
295 gl.push_back(toPixel(point.distance(), point.elevation(), bounds().x(), bounds().y()));
296
297 m_metrics_node->geometry()->allocate(static_cast<int>(gl.size()));
298 m_metrics_point_node->geometry()->allocate(static_cast<int>(gl.size()));
299 for(size_t i = 0; i < gl.size(); i++)
300 {
301 m_metrics_node->geometry()->vertexDataAsPoint2D()[i] = gl.at(i);
302 m_metrics_point_node->geometry()->vertexDataAsPoint2D()[i] = gl.at(i);
303 }
304 }
