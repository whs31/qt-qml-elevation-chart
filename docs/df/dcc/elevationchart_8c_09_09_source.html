166 void ElevationChartItem::updateBounds() noexcept
167 {
168 if(not route().valid())
169 return;
170
171 if(m_profile.empty())
172 return;
173
174 researcher()->researchIntersections(route().toGeoPath());
175
176 setBounds({ m_profile.back().distance(), 0 });
177 auto path = route().toGeoPath().path();
178 for(const auto& coordinate : path)
179 if(coordinate.altitude() > bounds().y())
180 setBounds({bounds().x(), static_cast<float>(coordinate.altitude())});
181
182 if(shrinkMode() == ShrinkToProfileHeight)
183 setBounds({bounds().x(), std::max(bounds().y(), std::max_element(m_profile.cbegin(), m_profile.cend(),
184 [](const ElevationPoint& a, const ElevationPoint& b){ return a.elevation() < b.elevation();
185 })->elevation())});
186
187 this->updateMetrics();
188 this->update();
189 }

191 void ElevationChartItem::updateMetrics() noexcept
192 {
193 bool matching_metrics = true;
194
195 QGeoPath correct_path;
196 QGeoPath route_path = route().toGeoPath();
197 if(route_path.isEmpty())
198 return;
199
200 auto velocities = route().velocities();
201
202 correct_path.addCoordinate(route_path.path().first());
203 for(int i = 1; i < route_path.path().size(); i++)
204 {
205 const bool fallback = velocities.at(i) == 0;
206 const auto dx = static_cast<float>(route_path.path().at(i).distanceTo(route_path.path().at(i - 1)));
207 const auto dy = static_cast<float>(route_path.path().at(i).altitude() - correct_path.path().at(i - 1).altitude());
208 const float dy_min = fallback ? metrics().rateOfDescend() * dx / metrics().fallbackVelocity() : metrics().rateOfDescend() * dx / velocities.at(i);
209 const float dy_max = fallback ? metrics().rateOfClimb() * dx / metrics().fallbackVelocity() : metrics().rateOfClimb() * dx / velocities.at(i);
210
211 auto result = route_path.path().at(i);
212
213 // correct case
214 if((dy >= 0 and dy < dy_max) or (dy <= 0 and abs(dy) < dy_min))
215 {
216 correct_path.addCoordinate(result);
217 continue;
218 }
219
220 // climbs too fast
221 else if(dy > 0 and dy > dy_max)
222 {
223 result.setAltitude(correct_path.path().at(i - 1).altitude() + dy_max);
224 matching_metrics = false;
225 }
226
227 // descends too fast
228 else if(dy < 0 and abs(dy) > dy_min)
229 {
230 result.setAltitude(correct_path.path().at(i - 1).altitude() - dy_min);
231 matching_metrics = false;
232 }
233
234 correct_path.addCoordinate(result);
235 }
236
237 if(not matching_metrics)
238 {
239 setMatchingMetrics(false);
240 m_metrics_path = correct_path;
241 }
242 else
243 setMatchingMetrics(true);
244 }

291 void ElevationChartItem::handleMetricsNode() noexcept
292 {
293 auto geometry1 = tree()[MetricsNode]->geometry();
294 auto geometry2 = tree()[MetricsPointNode]->geometry();
295
296 if(matchingMetrics())
297 {
298 /*
299 * окей, это не совсем очевидная херня, поэтому я прокомментирую будущим поколениям.
300 * если мы не будем очищать буфер ноды, то при отсутствии вызова обновления метрик в
301 * соответствии с маршрутом мы получим невалидные данные. если эта нода была бы отдельным
302 * компонентом в qml, все решилось бы visible: !matchingMetrics, но внутри одного итема
303 * единственный варик это вручную чистить буфер вот таким вызовом.
304 */
305 geometry1->allocate(0);
306 geometry2->allocate(0);
307 return;
308 }
309
310 vector<ElevationPoint> t_route;
311 auto prev_coord = m_metrics_path.path().front();
312 float distance = 0;
313 for(const auto& point : m_metrics_path.path())
314 {
315 distance += static_cast<float>(point.distanceTo(prev_coord));
316 t_route.emplace_back(distance, point.altitude());
317 prev_coord = point;
318 }
319
320 vector<QSGGeometry::Point2D> gl;
321
322 for(const auto& point : t_route)
323 gl.push_back(toPixel(point.distance(), point.elevation(), bounds().x(), bounds().y()));
324
325 geometry1->allocate(static_cast<int>(gl.size()));
326 geometry2->allocate(static_cast<int>(gl.size()));
327 for(size_t i = 0; i < gl.size(); i++)
328 {
329 geometry1->vertexDataAsPoint2D()[i] = gl.at(i);
330 geometry2->vertexDataAsPoint2D()[i] = gl.at(i);
331 }
332 }

334 void ElevationChartItem::handleIntersectionsNode() noexcept
335 {
336 if(m_intersections.empty())
337 return;
338
339 auto geometry = tree()[IntersectionsNode]->geometry();
340
341 vector<QSGGeometry::TexturedPoint2D> gl;
342 bool flip = false;
343 for(const auto& point : m_intersections)
344 {
345 if(flip)
346 {
347 gl.push_back(LPVL::utils::fromPoint2DBounded(toPixel(point.distance(), point.elevation(), bounds().x(), bounds().y()), width(), height()));
348 gl.push_back(LPVL::utils::fromPoint2DBounded({toPixelX(point.distance(), bounds().x()), static_cast<float>(height())}, width(), height()));
349 flip ^= 1;
350 }
351 else
352 {
353 gl.push_back(LPVL::utils::fromPoint2DBounded({toPixelX(point.distance(), bounds().x()), static_cast<float>(height())}, width(), height()));
354 gl.push_back(LPVL::utils::fromPoint2DBounded(toPixel(point.distance(), point.elevation(), bounds().x(), bounds().y()), width(), height()));
355 flip ^= 1;
356 }
357 }
358
359 if(gl.size() % 2 != 0)
360 gl.push_back(LPVL::utils::fromPoint2DBounded({toPixelX(m_intersections.back().distance(), bounds().x()), static_cast<float>(height())}, width(), height()));
361
362 geometry->allocate(static_cast<int>(gl.size()));
363 for(size_t i = 0; i < gl.size(); i++)
364 geometry->vertexDataAsTexturedPoint2D()[i] = gl.at(i);
365 }
