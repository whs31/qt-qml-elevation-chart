107 void ElevationChartItem::updateBounds() noexcept
108 {
109 if(not route().valid())
110 return;
111
112 setBounds({ m_profile.back().distance(), 0 });
113 auto path = route().toGeoPath().path();
114 for(const auto& coordinate : path)
115 if(coordinate.altitude() > bounds().y())
116 setBounds({bounds().x(), static_cast<float>(coordinate.altitude())});
117
118 if(shrinkMode() == ShrinkToProfileHeight)
119 setBounds({bounds().x(), std::max(bounds().y(), std::max_element(m_profile.cbegin(), m_profile.cend(),
120 [](const ElevationPoint& a, const ElevationPoint& b){ return a.elevation() < b.elevation();
121 })->elevation())});
122
123 this->updateMetrics();
124 this->update();
125 }

127 void ElevationChartItem::updateMetrics() noexcept
128 {
129 bool matching_metrics = true;
130
131 QGeoPath correct_path;
132 QGeoPath route_path = route().toGeoPath();
133 if(route_path.isEmpty())
134 return;
135
136 //bool allow_individual_speeds = not qFuzzyCompare(route().at(0).velocity(), 0);
137 auto velocities = route().velocities();
138
139 correct_path.addCoordinate(route_path.path().first());
140 for(int i = 1; i < route_path.path().size(); i++)
141 {
142 const bool fallback = velocities.at(i) == 0;
143 const auto dx = static_cast<float>(route_path.path().at(i).distanceTo(route_path.path().at(i - 1)));
144 const auto dy = static_cast<float>(route_path.path().at(i).altitude() - correct_path.path().at(i - 1).altitude());
145 const float dy_min = fallback ? metrics().rateOfDescend() * dx / metrics().fallbackVelocity() : metrics().rateOfDescend() * dx / velocities.at(i);
146 const float dy_max = fallback ? metrics().rateOfClimb() * dx / metrics().fallbackVelocity() : metrics().rateOfClimb() * dx / velocities.at(i);
147
148 auto result = route_path.path().at(i);
149
150 // correct case
151 if((dy >= 0 and dy < dy_max) or (dy <= 0 and abs(dy) < dy_min))
152 {
153 correct_path.addCoordinate(result);
154 continue;
155 }
156
157 // climbs too fast
158 else if(dy > 0 and dy > dy_max)
159 {
160 result.setAltitude(correct_path.path().at(i - 1).altitude() + dy_max);
161 matching_metrics = false;
162 }
163
164 // descends too fast
165 else if(dy < 0 and abs(dy) > dy_min)
166 {
167 result.setAltitude(correct_path.path().at(i - 1).altitude() - dy_min);
168 matching_metrics = false;
169 }
170
171 correct_path.addCoordinate(result);
172 }
173
174 if(not matching_metrics)
175 {
176 setMatchingMetrics(false);
177 m_metrics_path = correct_path;
178 }
179 else
180 setMatchingMetrics(true);
181 }

228 void ElevationChartItem::handleMetricsNode() noexcept
229 {
230 auto geometry1 = tree()[MetricsNode]->geometry();
231 auto geometry2 = tree()[MetricsPointNode]->geometry();
232
233 if(matchingMetrics())
234 {
235 /*
236 * окей, это не совсем очевидная херня, поэтому я прокомментирую будущим поколениям.
237 * если мы не будем очищать буфер ноды, то при отсутствии вызова обновления метрик в
238 * соответствии с маршрутом мы получим невалидные данные. если эта нода была бы отдельным
239 * компонентом в qml, все решилось бы visible: !matchingMetrics, но внутри одного итема
240 * единственный варик это вручную чистить буфер вот таким вызовом.
241 */
242 geometry1->allocate(0);
243 geometry2->allocate(0);
244 return;
245 }
246
247 vector<ElevationPoint> t_route;
248 auto prev_coord = m_metrics_path.path().front();
249 float distance = 0;
250 for(const auto& point : m_metrics_path.path())
251 {
252 distance += static_cast<float>(point.distanceTo(prev_coord));
253 t_route.emplace_back(distance, point.altitude());
254 prev_coord = point;
255 }
256
257 vector<QSGGeometry::Point2D> gl;
258
259 for(const auto& point : t_route)
260 gl.push_back(toPixel(point.distance(), point.elevation(), bounds().x(), bounds().y()));
261
262 geometry1->allocate(static_cast<int>(gl.size()));
263 geometry2->allocate(static_cast<int>(gl.size()));
264 for(size_t i = 0; i < gl.size(); i++)
265 {
266 geometry1->vertexDataAsPoint2D()[i] = gl.at(i);
267 geometry2->vertexDataAsPoint2D()[i] = gl.at(i);
268 }
269 }
