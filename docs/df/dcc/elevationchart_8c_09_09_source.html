37 QSGNode* ElevationChartItem::updatePaintNode(QSGNode* old_node, QQuickItem::UpdatePaintNodeData* unused)
38 {
39 (void)unused;
40
41 if(old_node == nullptr)
42 {
43 old_node = new QSGNode;
44
45 m_background_node = new QSGGeometryNode;
46 auto* background_material = new QSGFlatColorMaterial;
47 background_material->setColor(palette().background());
48 m_background_node->setMaterial(background_material);
49 auto* background_geometry = new QSGGeometry(QSGGeometry::defaultAttributes_Point2D(), 0, 0, QSGGeometry::UnsignedIntType);
50 background_geometry->setDrawingMode(QSGGeometry::DrawTriangles);
51 m_background_node->setGeometry(background_geometry);
52 m_background_node->setFlags(QSGNode::OwnsGeometry | QSGNode::OwnsMaterial);
53
54 m_profile_node = new QSGGeometryNode;
55 auto* profile_material = new QSGFlatColorMaterial;
56 profile_material->setColor(palette().overlay());
57 m_profile_node->setMaterial(profile_material);
58 auto* profile_geometry = new QSGGeometry(QSGGeometry::defaultAttributes_Point2D(), 0, 0, QSGGeometry::UnsignedIntType);
59 profile_geometry->setDrawingMode(GL_QUAD_STRIP);
60 m_profile_node->setGeometry(profile_geometry);
61 m_profile_node->setFlags(QSGNode::OwnsGeometry | QSGNode::OwnsMaterial);
62
63 m_route_node = new QSGGeometryNode;
64 auto* route_material = new QSGFlatColorMaterial;
65 route_material->setColor(palette().accent());
66 m_route_node->setMaterial(route_material);
67 auto* route_geometry = new QSGGeometry(QSGGeometry::defaultAttributes_Point2D(), 0, 0, QSGGeometry::UnsignedIntType);
68 route_geometry->setDrawingMode(QSGGeometry::DrawLineStrip);
69 route_geometry->setLineWidth(ROUTE_LINE_WIDTH);
70 m_route_node->setGeometry(route_geometry);
71 m_route_node->setFlags(QSGNode::OwnsGeometry | QSGNode::OwnsMaterial);
72
73 old_node->appendChildNode(m_background_node);
74 old_node->appendChildNode(m_profile_node);
75 old_node->appendChildNode(m_route_node);
76 }
77
78 if(m_require_recolor)
79 {
80 dynamic_cast<QSGFlatColorMaterial*>(m_background_node->material())->setColor(palette().background());
81 dynamic_cast<QSGFlatColorMaterial*>(m_profile_node->material())->setColor(palette().overlay());
82 dynamic_cast<QSGFlatColorMaterial*>(m_route_node->material())->setColor(palette().accent());
83 }
84
85 this->handleBackgroundNode();
86 this->handleProfileNode();
87 this->handleRouteNode();
88
89 if(m_require_recolor)
90 {
91 m_background_node->markDirty(QSGNode::DirtyGeometry | QSGNode::DirtyMaterial);
92 m_profile_node->markDirty(QSGNode::DirtyGeometry | QSGNode::DirtyMaterial);
93 m_route_node->markDirty(QSGNode::DirtyGeometry | QSGNode::DirtyMaterial);
94 this->fulfillRecolor();
95 }
96 else
97 {
98 m_background_node->markDirty(QSGNode::DirtyGeometry);
99 m_profile_node->markDirty(QSGNode::DirtyGeometry);
100 m_route_node->markDirty(QSGNode::DirtyGeometry);
101 }
102
103 old_node->markDirty(QSGNode::DirtyGeometry | QSGNode::DirtyMaterial);
104 return old_node;
105 }

127 void ElevationChartItem::updateBounds() noexcept
128 {
129 if(not route().valid())
130 return;
131
132 setBounds({ m_profile.back().distance(), 0 });
133 auto path = route().toGeoPath().path();
134 for(const auto& coordinate : path)
135 if(coordinate.altitude() > bounds().y())
136 setBounds({bounds().x(), static_cast<float>(coordinate.altitude())});
137
138 if(shrinkMode() == ShrinkToProfileHeight)
139 setBounds({bounds().x(), std::max(bounds().y(), std::max_element(m_profile.cbegin(), m_profile.cend(),
140 [](const ElevationPoint& a, const ElevationPoint& b){ return a.elevation() < b.elevation();
141 })->elevation())});
142
143 this->update();
144 }

146 void ElevationChartItem::handleBackgroundNode() noexcept
147 {
148 m_background_node->geometry()->allocate(6);
149 m_background_node->geometry()->vertexDataAsPoint2D()[0].set(0, 0);
150 m_background_node->geometry()->vertexDataAsPoint2D()[1].set(static_cast<float>(width()), 0);
151 m_background_node->geometry()->vertexDataAsPoint2D()[2].set(static_cast<float>(width()), static_cast<float>(height()));
152 m_background_node->geometry()->vertexDataAsPoint2D()[3].set(static_cast<float>(width()), static_cast<float>(height()));
153 m_background_node->geometry()->vertexDataAsPoint2D()[4].set(0, static_cast<float>(height()));
154 m_background_node->geometry()->vertexDataAsPoint2D()[5].set(0, 0);
155 }

157 void ElevationChartItem::handleProfileNode() noexcept
158 {
159 vector<QSGGeometry::Point2D> gl;
160 for(const auto& point : m_profile)
161 {
162 gl.push_back({toPixelX(point.distance(), bounds().x()), static_cast<float>(height())});
163 gl.push_back(toPixel(point.distance(), point.elevation(), bounds().x(), bounds().y()));
164 }
165
166 m_profile_node->geometry()->allocate(static_cast<int>(gl.size()));
167
168 for(size_t i = 0; i < gl.size(); i++)
169 m_profile_node->geometry()->vertexDataAsPoint2D()[i] = gl.at(i);
170 }

172 void ElevationChartItem::handleRouteNode() noexcept
173 {
174 vector<ElevationPoint> t_route = route().toElevationGraph();
175 vector<QSGGeometry::Point2D> gl;
176
177 for(const auto& point : t_route)
178 gl.push_back(toPixel(point.distance(), point.elevation(), bounds().x(), bounds().y()));
179
180 m_route_node->geometry()->allocate(static_cast<int>(gl.size()));
181 for(size_t i = 0; i < gl.size(); i++)
182 m_route_node->geometry()->vertexDataAsPoint2D()[i] = gl.at(i);
183 }
