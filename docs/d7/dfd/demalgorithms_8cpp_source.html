87int16_t DEM::elevation(const double latitude, const double longitude)
88{
89 double lat = latitude;
90 double lon = longitude;
91 if (latitude - std::floor(latitude) < 0.00001) {
92 lat = std::floor(latitude);
93 }
94 if (std::ceil(latitude) - latitude < 0.00001) {
95 lat = std::ceil(latitude);
96 }
97 if (longitude - std::floor(longitude) < 0.00001) {
98 lon = std::floor(longitude);
99 }
100 if (std::ceil(longitude) - longitude < 0.00001) {
101 lon = std::ceil(longitude);
102 }
103
104 try
105 {
106 return DEM::TileStorage::get()->elevation(lat, lon);
107 }
108 catch(std::runtime_error& x)
109 {
110 qCritical() << "<DEM> Exception caught:" << x.what();
111 throw std::runtime_error("DEM::elevation - access to non-created tile");
112 }
113}

121vector<pair<uint32_t, int16_t>> DEM::buildProfile(const QGeoPath& path, uint8_t discrete)
122{
123 loadTiles(path);
124 vector<pair<uint32_t, int16_t>> groundProfile;
125
126 if(path.size())
127 {
128 double distanceFromStart = 0;
129 QGeoCoordinate prevBasePointGeo = path.coordinateAt(0);
130
131 for(QGeoCoordinate point : path.path())
132 {
133
134 if(not groundProfile.empty())
135 {
136 double azimuth = prevBasePointGeo.azimuthTo(point);
137 double distanceFromPrevBasePoint = prevBasePointGeo.distanceTo(point);
138
139 double distance = discrete;
140 QGeoCoordinate prevDeltaPointGeo = prevBasePointGeo;
141 while(distance < distanceFromPrevBasePoint)
142 {
143 QGeoCoordinate deltaPointGeo = prevBasePointGeo.atDistanceAndAzimuth(distance, azimuth);
144 auto a = deltaPointGeo.latitude();
145 auto b = deltaPointGeo.longitude();
146 auto c = prevDeltaPointGeo.latitude();
147 auto d = prevDeltaPointGeo.longitude();
148 auto e = point.latitude();
149 auto f = point.longitude();
150
151
152 try
153 {
154 deltaPointGeo.setAltitude(DEM::elevation(deltaPointGeo.latitude(), deltaPointGeo.longitude()));
155 }
156 catch(std::runtime_error& x)
157 {
158 qCritical() << "<DEM> Exception caught:" << x.what();
159 return {};
160 }
161 if(prevDeltaPointGeo.altitude() != deltaPointGeo.altitude())
162 {
163 if(prevDeltaPointGeo.altitude() > deltaPointGeo.altitude())
164 groundProfile.emplace_back(distance - discrete + distanceFromStart, prevDeltaPointGeo.altitude());
165 else
166 groundProfile.emplace_back(distance + distanceFromStart, deltaPointGeo.altitude());
167 }
168
169 prevDeltaPointGeo = deltaPointGeo;
170 distance += discrete;
171 }
172
173 distanceFromStart += distanceFromPrevBasePoint;
174 }
175
176 try
177 {
178 point.setAltitude(DEM::elevation(point.latitude(), point.longitude()));
179 }
180 catch(std::runtime_error& x)
181 {
182 qCritical() << "<DEM> Exception caught:" << x.what();
183 return {};
184 }
185
186 groundProfile.emplace_back(distanceFromStart, point.altitude());
187 prevBasePointGeo = point;
188 }
189 }
190 return groundProfile;
191}

197void DEM::loadTiles(const QGeoPath &_path)
198{
199 if (_path.size() == 0)
200 return;
201
202 if (_path.size() == 1)
203 {
204 loadTile(std::floor(_path.coordinateAt(0).latitude()), std::floor(_path.coordinateAt(0).longitude()));
205 return;
206 }
207
208 int8_t minLatitude{};
209 int8_t maxLatitude{};
210 int16_t minLongitude{};
211 int16_t maxLongitude{};
212
213 for(int i = 1; i < _path.size(); ++i)
214 {
215 QGeoCoordinate coord1 = _path.coordinateAt(i - 1);
216 QGeoCoordinate coord2 = _path.coordinateAt(i);
217
218 minLatitude = coord1.latitude() < coord2.latitude() ? static_cast<int8_t>(coord1.latitude()) : static_cast<int8_t>(coord2.latitude());
219 maxLatitude = coord1.latitude() < coord2.latitude() ? static_cast<int8_t>(coord2.latitude()) : static_cast<int8_t>(coord1.latitude());
220 minLongitude = coord1.longitude() < coord2.longitude() ? static_cast<int16_t>(coord1.longitude()) : static_cast<int16_t>(coord2.longitude());
221 maxLongitude = coord1.longitude() < coord2.longitude() ? static_cast<int16_t>(coord2.longitude()) : static_cast<int16_t>(coord1.longitude());
222
223 loadTiles(minLatitude, minLongitude, maxLatitude, maxLongitude);
224 }
225}
