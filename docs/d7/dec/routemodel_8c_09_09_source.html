14 QVariant RouteModel::data(const QModelIndex& index, int role) const
15 {
16 if(not index.isValid())

19 {
20 case Index: return index.row();
21 case Altitude: return QVariant::fromValue(m_storage.at(index.row()).elevation());
22 case Distance: return QVariant::fromValue(m_storage.at(index.row()).distance());
23 case Valid: return QVariant::fromValue(m_storage.at(index.row()).valid());

30 bool RouteModel::setData(const QModelIndex& index, const QVariant& value, int role)
31 {

35 switch(role)
36 {
37 case Index: return false;

40 case Valid: m_storage.at(index.row()).setValid(value.toBool()); break;
41 case Intersects: m_storage.at(index.row()).setIntersects(value.toBool()); break;
42
43 default: return false;
44
45 emit dataChanged(index, index, {role});

47 }
